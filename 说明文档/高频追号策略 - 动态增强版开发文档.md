\# 特肖高频追号策略 - 动态增强版开发文档

&#x20;



\## 1. 项目概述



\### 1.1 核心功能

基于历史开奖数据，动态判断市场行情（热市/冷市），自适应调整统计期数、高频阈值、追号期数，输出4个推荐生肖，并提供信心评分和操作建议。



\### 1.2 版本历史



\| 版本 | 主要特性 |

\|------|----------|

\| v1.0 | 固定27期、固定阈值≥3、固定追号3期 |

\| v2.0 | 动态参数切换 + 连错风控 |

\| v2.1 | 增加信心评分 + 操作建议 |

// core/market.js

const config = require('../config/strategy.config');



function getMarketCondition(periods) {

  // periods: PeriodRecord\[]，至少27期

  const lookback = config.marketRules.lookbackPeriods;

  

  // 1. 计算变化率（相邻期数不同的比例）

  let changeCount = 0;

  for (let i = 1; i < periods.length; i++) {

    if (periods\[i].zodiac !== periods\[i-1].zodiac) changeCount++;

  }

  const changeRate = changeCount / (periods.length - 1);

  

  // 2. 检查上期是否重复

  const lastRepeat = periods\[periods.length-1].zodiac === periods\[periods.length-2].zodiac;

  

  // 3. 计算最近10期特肖种类数

  const lastN = periods.slice(-lookback);

  const variety = new Set(lastN.map(p => p.zodiac)).size;

  

  let hotScore = 0;

  if (changeRate >= config.marketRules.changeRateThreshold) hotScore++;

  if (lastRepeat) hotScore++;

  if (variety <= config.marketRules.varietyThreshold) hotScore++;

  

  return hotScore >= 2 ? 'hot' : 'cold';

}



function getPeriodLength(market) {

  return config.periodLength\[market] || config.default.periodLength;

}



function getThreshold(market) {

  return config.threshold\[market] || config.default.threshold;

}



function getMaxAttempts(market) {

  return config.maxAttempts\[market] || config.default.maxAttempts;

}



module.exports = {

  getMarketCondition,

  getPeriodLength,

  getThreshold,

  getMaxAttempts

};

// core/statistics.js

function getRecentPeriods(data, count, excludeLast = true) {

  let result = \[...data];

  if (excludeLast && result.length > 0) {

    result = result.slice(0, -1);

  }

  return result.slice(-count);

}



function countZodiacFrequency(periods) {

  const freq = new Map();

  for (const p of periods) {

    freq.set(p.zodiac, (freq.get(p.zodiac) || 0) + 1);

  }

  return freq;

}



function sortByFrequencyAndRecency(periods, freqMap) {

  const lastIndex = new Map();

  for (let i = periods.length - 1; i >= 0; i--) {

    const zod = periods\[i].zodiac;

    if (!lastIndex.has(zod)) {

      lastIndex.set(zod, periods.length - 1 - i);

    }

  }

  

  const zodiacs = \[...new Set(periods.map(p => p.zodiac))];

  return zodiacs.sort((a, b) => {

    const diff = (freqMap.get(b) || 0) - (freqMap.get(a) || 0);

    if (diff !== 0) return diff;

    return (lastIndex.get(a) || Infinity) - (lastIndex.get(b) || Infinity);

  });

}



function getRecentSet(periods, lookbackCount) {

  const recent = periods.slice(-lookbackCount);

  return new Set(recent.map(p => p.zodiac));

}



module.exports = {

  getRecentPeriods,

  countZodiacFrequency,

  sortByFrequencyAndRecency,

  getRecentSet

};

\### 4.3 推荐算法模块



\`\`\`javascript

// core/recommend.js

const {

  getRecentPeriods,

  countZodiacFrequency,

  sortByFrequencyAndRecency,

  getRecentSet

} = require('./statistics');

const { getMarketCondition, getPeriodLength, getThreshold } = require('./market');



function recommendNext(history) {

  if (history.length < 25) {

    throw new Error('至少需要25期历史数据');

  }

  

  // 1. 判断行情

  const market = getMarketCondition(history.slice(-27));

  

  // 2. 获取动态参数

  const periodLen = getPeriodLength(market);

  const threshold = getThreshold(market);

  

  // 3. 取最近N期（剔除上期）

  const recent = getRecentPeriods(history, periodLen, true);

  

  // 4. 统计并排序

  const freq = countZodiacFrequency(recent);

  const sorted = sortByFrequencyAndRecency(recent, freq);

  

  // 5. 选取高频

  const highFreq = sorted.filter(z => (freq.get(z) || 0) >= threshold);

  

  // 6. 补足至4个

  let recommendation;

  if (highFreq.length >= 4) {

    recommendation = highFreq.slice(0, 4);

  } else {

    const fallback = sorted.filter(z => (freq.get(z) || 0) === 2);

    const recent10Set = getRecentSet(recent, 10);

    const priority = fallback.filter(z => recent10Set.has(z));

    const needed = 4 - highFreq.length;

    

    if (priority.length >= needed) {

      recommendation = highFreq.concat(priority.slice(0, needed));

    } else {

      const remaining = needed - priority.length;

      recommendation = highFreq.concat(priority, fallback.slice(0, remaining));

    }

  }

  

  return {

    recommendation,

    market,

    periodLen,

    threshold,

    freq: Object.fromEntries(freq)

  };

}



module.exports = { recommendNext };

function getConfidenceScore(market, highFreqCount, recentHitRate) {

  const weights = config.confidenceWeights;

  let score = 0;

  

  // 行情加分

  if (market === 'hot') score += weights.market;

  else if (market === 'cold') score += weights.market \* 0.5;

  else score += weights.market \* 0.75;

  

  // 高频数量加分

  const highFreqScore = Math.min(highFreqCount, 4) / 4 \* weights.highFreq;

  score += highFreqScore;

  

  // 命中率加分

  score += recentHitRate \* weights.hitRate;

  

  return Math.min(100, Math.max(0, Math.round(score)));

}



function getSuggestion(score, isPaused) {

  if (isPaused) {

    return {

      action: '🛑 暂停交易',

      color: 'red',

      reason: '连续错2组，等待企稳'

    };

  }

  

  const thresholds = config.suggestionThresholds;

  if (score >= thresholds.strong) {

    return {

      action: '✅ 建议出手',

      color: 'green',

      reason: '信号强烈，正常追号'

    };

      action: '⚠️ 谨慎出手',

      color: 'yellow',

      reason: '信号一般，可降低仓位'

\### 4.5 风险控制器



\`\`\`javascript

// risk/riskController.js

const config = require('../config/strategy.config');



class RiskController {

  constructor() {

    this.consecutiveMissGroups = 0;

    this.isPaused = false;

    this.observationPeriods = \[];

  }

  

  recordGroupResult(isWin) {

    if (this.isPaused) return;

    

    if (!isWin) {

      this.consecutiveMissGroups++;

      if (this.consecutiveMissGroups >= config.risk.consecutiveMissGroups) {

        this.isPaused = true;

        return true; // 触发暂停

      }

    } else {

      this.consecutiveMissGroups = 0;

    }

    return false;

  }

  

  observePeriod(zodiac) {

    if (!this.isPaused) return false;

    

    this.observationPeriods.push(zodiac);

    if (this.observationPeriods.length > config.risk.observationPeriods) {

      this.observationPeriods.shift();

    }

    

    if (this.observationPeriods.length === config.risk.observationPeriods) {

      const unique = new Set(this.observationPeriods);

      if (unique.size === config.risk.observationPeriods) {

        this.isPaused = false;

        this.consecutiveMissGroups = 0;

        this.observationPeriods = \[];

        return true; // 企稳恢复

      }

    }

    return false;

  }

  

  canRecommend() {

    return !this.isPaused;

  }

  

  getStatus() {

    return {

      isPaused: this.isPaused,

      consecutiveMissGroups: this.consecutiveMissGroups

    };

  }

}



module.exports = RiskController;

\### 4.6 追中即停控制器



\`\`\`javascript

// core/chaseController.js

class ChaseController {

  constructor(riskController) {

    this.currentGroup = null;

    this.remainingAttempts = 0;

    this.riskController = riskController;

  }

  

  startNewGroup(recommendation, maxAttempts) {

    this.currentGroup = recommendation;

    this.remainingAttempts = maxAttempts;

  }

  

  checkResult(actualZodiac) {

    if (!this.currentGroup) return { status: 'no\_active\_group' };

    

    const isWin = this.currentGroup.includes(actualZodiac);

    this.remainingAttempts--;

    

    if (isWin) {

      this.riskController.recordGroupResult(true);

      this.currentGroup = null;

      return { status: 'win', remainingAttempts: 0 };

    }

    

    if (this.remainingAttempts <= 0) {

      const triggered = this.riskController.recordGroupResult(false);

      this.currentGroup = null;

      return { 

        status: 'reset', 

        remainingAttempts: 0,

        triggeredPause: triggered

      };

    }

    

    return { status: 'continue', remainingAttempts: this.remainingAttempts };

  }

  

  needRecalculate() {

    return this.currentGroup === null;

  }

  

  getActiveGroup() {

    return {

      recommendation: this.currentGroup,

      remainingAttempts: this.remainingAttempts

    };

  }

}



module.exports = ChaseController;

\## 5. 主入口



\`\`\`javascript

// index.js

const { recommendNext } = require('./core/recommend');

const { getMaxAttempts } = require('./core/market');

const { calculateRecentHitRate, getConfidenceScore, getSuggestion } = require('./core/confidence');

const RiskController = require('./risk/riskController');

const ChaseController = require('./core/chaseController');



class StrategyEngine {

  constructor() {

    this.riskController = new RiskController();

    this.chaseController = new ChaseController(this.riskController);

    this.lastRecommendation = null;

  }

  

  // 每期调用

  onNewPeriod(history, actualZodiac = null) {

    // 1. 若有实际开奖，先验证

    if (actualZodiac && !this.chaseController.needRecalculate()) {

      const result = this.chaseController.checkResult(actualZodiac);

      if (result.status === 'continue') {

        return { 

          action: 'continue\_chase',

          remainingAttempts: result.remainingAttempts,

          activeGroup: this.chaseController.getActiveGroup()

        };

      }

    }

    

    // 2. 风控观察（若处于暂停状态）

    if (!this.riskController.canRecommend()) {

      if (actualZodiac) {

        const recovered = this.riskController.observePeriod(actualZodiac);

        if (!recovered) {

          return {

            action: 'paused',

            message: '风控暂停中，等待行情企稳',

            riskStatus: this.riskController.getStatus()

          };

        }

      } else {

        return {

          action: 'paused',

          message: '风控暂停中，等待行情企稳',

          riskStatus: this.riskController.getStatus()

        };

      }

    }

    

    // 3. 计算新推荐

    const recResult = recommendNext(history);

    const maxAttempts = getMaxAttempts(recResult.market);

    const hitRate = calculateRecentHitRate(history);

    const confidenceScore = getConfidenceScore(

      recResult.market,

      recResult.recommendation.length,

      hitRate

    );

    const suggestion = getSuggestion(confidenceScore, false);

    

    // 4. 开始新追号

    this.chaseController.startNewGroup(recResult.recommendation, maxAttempts);

    this.lastRecommendation = recResult;

    

    return {

      action: 'new\_recommendation',

      recommendation: recResult.recommendation,

      market: recResult.market,

      periodLen: recResult.periodLen,

      threshold: recResult.threshold,

      maxAttempts: maxAttempts,

      confidenceScore: confidenceScore,

      suggestion: suggestion,

      riskStatus: this.riskController.getStatus()

    };

  }

  

  getStatus() {

    return {

      isPaused: this.riskController.canRecommend() ? false : true,

      activeGroup: this.chaseController.getActiveGroup(),

      lastRecommendation: this.lastRecommendation

    };

  }

}



module.exports = StrategyEngine;
